func sendDataToDest(data []byte, dst *string) {
        conn, err := net.Dial("tcp", *dst)
        if err != nil {
                fmt.Fprintf(os.Stderr, "Fatal error: %s", err.Error())
                return
        }
        n, err := conn.Write(data)
        if err != nil {
                fmt.Fprintf(os.Stderr, "Fatal error: %s", err.Error())
                return
        }
        fmt.Println("Sent " + strconv.Itoa(n) + " bytes")
}
